#include <iostream>
#include <queue>
using namespace  std;

int getkthsmallele( int arr[], int n, int k){

  //create max heap
  priority_queue<int> pq;

  //insert starting ke k.elements of arrray in heap
  for(int i=0 ; i<k; i++){
    pq.push(arr[i]);
    
  }

  //for reamianig ele, insert only thiose ele that are less than top ele of max heap created 
  for(int i=k; i<n;i++){
     int ele= arr[i];
    if(ele < pq.top()){

      pq.pop();
      pq.push(ele);
      
    }
  }

  int ans= pq.top();
  return ans;
}

int main() {
  int arr[]={10, 5,9,11,6};
  int n= 5;
  int k= 2;
  int ans = getkthsmallele(arr, n, k);
  cout<< "ans is: "<< ans;
  return 0;
}
